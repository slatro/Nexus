# Kurulum

```console
# güncelleme
sudo apt update -y && sudo apt upgrade -y
sudo apt install cmake
sudo apt install build-essential
```

#

```console
# rustup kurulumu

# 1 seçeneğini seçiyoruz
curl --proto '=https' --tlsv1.2 -sSf https://sh.rustup.rs | sh

# rust kurulumu tamamlandıktan sonra
. "$HOME/.cargo/env"
rustup target add riscv32i-unknown-none-elf
```

#

```console
# nexus tool kurulumu

# burası biraz uzun sürer - hatalar görürseniz sorun yok.
cargo install --git https://github.com/nexus-xyz/nexus-zkvm nexus-tools --tag 'v1.0.0'

# nexus oluşturuyoruz
cargo nexus new nexus-project

# main.rs değiştireceğiz
cd nexus-project
```

#

```console

# bu dosyaya giriyoruz
nano ./src/main.rs
# içersindeki kodları silip aşağıdaki bloğu girin
```

```console
#![no_std]
#![no_main]

fn fib(n: u32) -> u32 {
    match n {
        0 => 0,
        1 => 1,
        _ => fib(n - 1) + fib(n - 2),
    }
}

#[nexus_rt::main]
fn main() {
    let n = 7;
    let result = fib(n);
    assert_eq!(result, 13);
}
```

#

```console
# contratı run edelim
cargo nexus run
cargo nexus run -v

# prove etmeseini bekleyelim işlemlerin
cargo nexus prove

# verify işlemini tamamlayalım
cargo nexus verify
```
